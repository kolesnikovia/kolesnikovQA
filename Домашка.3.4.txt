1.options=(
	'1'
	"2"
	'3'
	'4'
	'5'
) 
-Создается 5 опций

2.echo 'Please choose an option'-показывает выбранную опцию

select input in "${options[@]}"; do
	case "$input" in
		"${options[0]}")
			ls
			;;
3.Опция "0" показывет все файлы и папки в рабочей директории
		"${options[1]}")
			date $'+Time: %T\nDate: %D'
			;;
4.Опция "1" показывает дату и время
		"${options[2]}")
			read -p 'What file do you wish to check? ' finput
			if [[ -d $finput ]]; then
				format='%s is a directory.\n'
			elif [[ -f $finput ]]; then
				format='%s is a file.\n'
			else
				format='%s does not exist.\n'
			fi
			printf "$format" "$finput"
			;;
5.Опция "2" считывает какой файл нужно проверить, -d проверяет является ли директорий,
 если да-то записывает в формат что формат=директория, -f проверяет является ли файлом,
 если да-то записывает в формат что формат=файл,
если ни файл, ни директория, то пишет, что формат не существует;
выводит сообщение определенного формата введенного объекта
		"${options[3]}")
			read -p 'Please enter filename: ' binput
			cp "$binput"{,.bak}
			;;
6.Опция "3" считывает название файла и копирует в буфер обмена
		"${options[4]}")
			break
			;;
		*)
			echo 'Not a valid option.'
			;;
7.Опция "4" никаких действий не происходит, выдается сообщение, что опция невалидная
	esac
done

Заключение:
Опция "0" не работает( не валидна)
Опция "1" выполняет работу опции 0 т.е показывает все файлы
Опция "2" выполняет функцию опции "1" 
Опция "3" выполняет функцию опции "2" 
Опция "4" выполняет опции "3"
